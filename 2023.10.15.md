```

{

```

## 다중 do while문

```
do
{
  do
  {
    블록;
  } while (조건식);
} while (조건식);
```

### 실습문제

```
#include <stdio.h>
int main(void)
{
	int num1=2, num2=1;

	while (num1 <= 9)
	{
		while (num2 <= 9)
		{
			printf("%2d * %d = %d", num2, num1, num1 * num2);
			num2 += 1;
		}

		num1 += 1;
		num2 = 1;
		printf("\n");
	}

	return 0;
}
```

### 실습문제

```
#include <stdio.h>
int main(void)
{
	int num, i = 1, j, fac;

	printf("1부터 팩토리얼을 계산할 정수 입력: ");
	scanf_s("%d", &num);

	do
	{
		fac = 1;
		j = 1;
		do
		{
			fac = fac * j;
			j += 1;
		} while (j <= i);

		printf("%d!은 %d\n", i, fac);
		i += 1;

	} while (i <= num);

	return 0;
}
```

```

### 예제 8-3 

```
#include <stdio.h>
int main(void)
{
	int jul = 6, i;

	while (jul>=1)
	{
		i = 1;
		while (i <= jul)
		{
			printf("#");
			i += 1;
		}

		printf("\n");
		jul -= 1;
	}

	return 0;
}
